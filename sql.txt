import tkinter as tk
import psycopg2

# Conecta ao banco de dados
conn = psycopg2.connect(
    host="seu_host",
    database="seu_banco_de_dados",
    user="seu_usuario",
    password="sua_senha"
)

# Cria uma janela Tkinter
window = tk.Tk()
window.title("CRUD")

# Função para inserir um novo cliente
def inserir_cliente():
    nome = entry_nome.get()
    telefone = entry_telefone.get()
    endereco = entry_endereco.get()

    cursor = conn.cursor()
    cursor.execute("INSERT INTO Clientes (nome, telefone, endereco) VALUES (%s, %s, %s)", (nome, telefone, endereco))
    conn.commit()
    cursor.close()
    limpar_campos()

# Função para exibir todos os clientes
def exibir_clientes():
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM Clientes")
    rows = cursor.fetchall()
    text_box.delete(1.0, tk.END)

    for row in rows:
        text_box.insert(tk.END, f"ID: {row[0]}\nNome: {row[1]}\nTelefone: {row[2]}\nEndereço: {row[3]}\n\n")

    cursor.close()

# Função para limpar os campos de entrada
def limpar_campos():
    entry_nome.delete(0, tk.END)
    entry_telefone.delete(0, tk.END)
    entry_endereco.delete(0, tk.END)

# Labels e campos de entrada
label_nome = tk.Label(window, text="Nome:")
label_nome.pack()
entry_nome = tk.Entry(window)
entry_nome.pack()

label_telefone = tk.Label(window, text="Telefone:")
label_telefone.pack()
entry_telefone = tk.Entry(window)
entry_telefone.pack()

label_endereco = tk.Label(window, text="Endereço:")
label_endereco.pack()
entry_endereco = tk.Entry(window)
entry_endereco.pack()

# Botões
btn_inserir = tk.Button(window, text="Inserir Cliente", command=inserir_cliente)
btn_inserir.pack()

btn_exibir = tk.Button(window, text="Exibir Clientes", command=exibir_clientes)
btn_exibir.pack()

# Text box para exibir os clientes
text_box = tk.Text(window)
text_box.pack()

# Executa a janela principal
window.mainloop()

# Fecha a conexão com o banco de dados ao fechar a janela
conn.close()
